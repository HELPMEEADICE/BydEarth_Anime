country_event = {
	id = centralasia_event.10
	title = centralasia_event.10.t
	desc = centralasia_event.10.d
	
	is_triggered_only = yes
	
	option = { 
		name = centralasia_event_option.10.a
		if = {
			limit = {
				check_variable = { undecided_ticker > social_democrat_ticker }
				check_variable = { undecided_ticker > conservative_ticker }
				check_variable = { undecided_ticker > supremacist_ticker }
			}
			country_event = {
				id = centralasia_event.11
			}
		}
		else_if = {
			limit = {
				check_variable = { social_democrat_ticker > undecided_ticker }
				check_variable = { social_democrat_ticker > conservative_ticker }
				check_variable = { social_democrat_ticker > supremacist_ticker }
			}
			country_event = {
				id = centralasia_event.12
			}
		}
		else_if = {
			limit = {
				check_variable = { conservative_ticker > undecided_ticker }
				check_variable = { conservative_ticker > social_democrat_ticker }
				check_variable = { conservative_ticker > supremacist_ticker }
			}
			country_event = {
				id = centralasia_event.13
			}
		}
		else_if = {
			limit = {
				check_variable = { supremacist_ticker > undecided_ticker }
				check_variable = { supremacist_ticker > social_democrat_ticker }
				check_variable = { supremacist_ticker > conservative_ticker }
			}
			country_event = {
				id = centralasia_event.14
			}
		}
	}
}
